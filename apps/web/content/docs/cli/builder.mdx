## Usage

<Tabs items={['bun', 'pnpm', 'npm']}>
  <Tab value="bun">
    ```bash
    bun create better-t-stack@latest builder
    ```
  </Tab>
  <Tab value="pnpm">
    ```bash
    pnpm create better-t-stack@latest builder
    ```
  </Tab>
  <Tab value="npm">
    ```bash
    npx create-better-t-stack@latest builder
    ```
  </Tab>
</Tabs>
